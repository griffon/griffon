package org.codehaus.griffon.runtime.core;

import griffon.core.GriffonApplication;
import griffon.core.injection.Module;

import javax.annotation.Nonnull;
import java.util.ArrayList;
import java.util.List;
import java.util.ServiceLoader;

/**
 * @author Andres Almiray
 * @since 2.0.0
 */
public class DefaultApplicationBootstrapper extends AbstractApplicationBootstrapper {
    public DefaultApplicationBootstrapper(@Nonnull GriffonApplication application) {
        super(application);
    }

    @Nonnull
    @Override
    protected List<Module> loadModules() {
        List<Module> moduleInstances = new ArrayList<>();
        ServiceLoader<Module> serviceLoader = ServiceLoader.load(Module.class);
        for (Module module : serviceLoader) {
            moduleInstances.add(module);
        }
        return moduleInstances;
    }
}
