314     public boolean canShutdown() {
315         event(ApplicationEvent.SHUTDOWN_REQUESTED, asList(this));
316         synchronized (shutdownLock) {
317             for (ShutdownHandler handler : shutdownHandlers) {
318                 if (!handler.canShutdown(this)) {
319                     event(ApplicationEvent.SHUTDOWN_ABORTED, asList(this));
320                     if (log.isDebugEnabled()) {
321                         try {
322                             log.debug("Shutdown aborted by " + handler);
323                         } catch (UnsupportedOperationException uoe) {
324                             log.debug("Shutdown aborted by a handler");
325                         }
326                     }
327                     return false;
328                 }
329             }
330         }
331         return true;
332     }

334     public boolean shutdown() {
335         // avoids reentrant calls to shutdown()
336         // once permission to quit has been granted
337         if (getPhase() == ApplicationPhase.SHUTDOWN) return false;
338 
339         if (!canShutdown()) return false;
340         log.info("Shutdown is in process");
341 
342         // signal that shutdown is in process
343         setPhase(ApplicationPhase.SHUTDOWN);
344 
345         // stage 1 - alert all app event handlers
346         // wait for all handlers to complete before proceeding
347         // with stage #2 if and only if the current thread is
348         // the ui thread
349         log.debug("Shutdown stage 1: notify all event listeners");
350         if (getEventRouter().isEventPublishingEnabled()) {
351             final CountDownLatch latch = new CountDownLatch(getUIThreadManager().isUIThread() ? 1 : 0);
352             getEventRouter().addEventListener(ApplicationEvent.SHUTDOWN_START.getName(), new CallableWithArgs<Void>() {
353                 @Override
354                 @Nullable
355                 public Void call(@Nullable Object... args) {
356                     latch.countDown();
357                     return null;
358                 }
359             });
360             event(ApplicationEvent.SHUTDOWN_START, asList(this));
361             try {
362                 latch.await();
363             } catch (InterruptedException e) {
364                 // ignore
365             }
366         }
367 
368         // stage 2 - alert all shutdown handlers
369         log.debug("Shutdown stage 2: notify all shutdown handlers");
370         synchronized (shutdownLock) {
371             for (ShutdownHandler handler : shutdownHandlers) {
372                 handler.onShutdown(this);
373             }
374         }
375 
376         // stage 3 - destroy all mvc groups
377         log.debug("Shutdown stage 3: destroy all MVC groups");
378         List<String> mvcIds = new ArrayList<>();
379         mvcIds.addAll(getMvcGroupManager().getGroups().keySet());
380         for (String id : mvcIds) {
381             getMvcGroupManager().destroyMVCGroup(id);
382         }
383 
384         // stage 4 - call shutdown script
385         log.debug("Shutdown stage 4: execute Shutdown script");
386         getApplicationConfigurer().runLifecycleHandler(Lifecycle.SHUTDOWN);
387 
388         injector.getInstance(ExecutorServiceManager.class).shutdownAll();
389         injector.close();
390 
391         return true;
392     }

394     @SuppressWarnings("unchecked")
395     public void startup() {
396         if (getPhase() != ApplicationPhase.INITIALIZE) return;
397 
398         setPhase(ApplicationPhase.STARTUP);
399         event(ApplicationEvent.STARTUP_START, asList(this));
400 
401         Object startupGroups = getConfiguration().get("application.startupGroups", null);
402         if (startupGroups instanceof List) {
403             if (log.isInfoEnabled()) {
404                 log.info("Initializing all startup groups: " + startupGroups);
405             }
406 
407             for (String groupName : (List<String>) startupGroups) {
408                 getMvcGroupManager().createMVCGroup(groupName);
409             }
410         } else if (startupGroups != null && startupGroups.getClass().isArray()) {
411             Object[] groups = (Object[]) startupGroups;
412             if (log.isInfoEnabled()) {
413                 log.info("Initializing all startup groups: " + Arrays.toString(groups));
414             }
415 
416             for (Object groupName : groups) {
417                 getMvcGroupManager().createMVCGroup(String.valueOf(groupName));
418             }
419         }
420 
421         getApplicationConfigurer().runLifecycleHandler(Lifecycle.STARTUP);
422 
423         event(ApplicationEvent.STARTUP_END, asList(this));
424     }
