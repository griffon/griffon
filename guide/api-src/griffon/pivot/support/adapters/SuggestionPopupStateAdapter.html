16 package griffon.pivot.support.adapters;
17 
18 import griffon.core.CallableWithArgs;
19 import org.apache.pivot.util.Vote;
20 
21 /**
22  * @author Andres Almiray
23  * @since 2.0.0
24  */
25 public class SuggestionPopupStateAdapter implements GriffonPivotAdapter, org.apache.pivot.wtk.SuggestionPopupStateListener {
26     private CallableWithArgs<Vote> previewSuggestionPopupClose;
27     private CallableWithArgs<Void> suggestionPopupCloseVetoed;
28     private CallableWithArgs<Void> suggestionPopupClosed;
29 
30     public CallableWithArgs<Vote> getPreviewSuggestionPopupClose() {
31         return this.previewSuggestionPopupClose;
32     }
33 
34     public CallableWithArgs<Void> getSuggestionPopupCloseVetoed() {
35         return this.suggestionPopupCloseVetoed;
36     }
37 
38     public CallableWithArgs<Void> getSuggestionPopupClosed() {
39         return this.suggestionPopupClosed;
40     }
41 
42 
43     public void setPreviewSuggestionPopupClose(CallableWithArgs<Vote> previewSuggestionPopupClose) {
44         this.previewSuggestionPopupClose = previewSuggestionPopupClose;
45     }
46 
47     public void setSuggestionPopupCloseVetoed(CallableWithArgs<Void> suggestionPopupCloseVetoed) {
48         this.suggestionPopupCloseVetoed = suggestionPopupCloseVetoed;
49     }
50 
51     public void setSuggestionPopupClosed(CallableWithArgs<Void> suggestionPopupClosed) {
52         this.suggestionPopupClosed = suggestionPopupClosed;
53     }
54 
55 
56     public org.apache.pivot.util.Vote previewSuggestionPopupClose(org.apache.pivot.wtk.SuggestionPopup arg0, boolean arg1) {
57         if (previewSuggestionPopupClose != null) {
58             return previewSuggestionPopupClose.call(arg0, arg1);
59         }
60         return Vote.APPROVE;
61     }
62 
63     public void suggestionPopupCloseVetoed(org.apache.pivot.wtk.SuggestionPopup arg0, org.apache.pivot.util.Vote arg1) {
64         if (suggestionPopupCloseVetoed != null) {
65             suggestionPopupCloseVetoed.call(arg0, arg1);
66         }
67     }
68 
69     public void suggestionPopupClosed(org.apache.pivot.wtk.SuggestionPopup arg0) {
70         if (suggestionPopupClosed != null) {
71             suggestionPopupClosed.call(arg0);
72         }
73     }
74 
75 }
